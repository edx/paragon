export const pageQuery = graphql`
{
  textUtilities: allCssUtilityClasses(
    filter: {isUtility: {eq: true}, declarations: {regex: "/^(?!color)/"}, selector: {regex: "/(^text)|(^font)/"}},
    sort: {fields: selector, order: ASC}
  ) {
    nodes {
      selector
      declarations
    }
  }
}
`;
