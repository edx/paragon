export const pageQuery = graphql`
{
  allCssUtilityClasses(
    filter: {declarations: {regex: "/color/"}, isUtility: {eq: true}},
    sort: {fields: selector, order: ASC}
  ) {
    nodes {
      selector
      declarations
    }
    distinct(field: selector)
  }
}
`;
