##### Basic Usage

```jsx live
<Form>
  <Form.Row>
    <Form.Group as={Col} controlId="formGridEmail">
      <Form.Label>Email</Form.Label>
      <Form.Control type="email" placeholder="Enter email" aria-describedby="emailHelpBlock" />
      <Form.Text id="emailHelpBlock" muted>
        This is the email that we'll send emails to.
      </Form.Text>
    </Form.Group>

    <Form.Group as={Col} controlId="formGridPassword">
      <Form.Label>Password</Form.Label>
      <Form.Control className="is-invalid" type="password" placeholder="Password" aria-describedby="passwordFeedback" />
      <Form.Control.Feedback id="passwordFeedback" type="invalid">
        We don't like that password. Try a different one.
      </Form.Control.Feedback>
    </Form.Group>
  </Form.Row>

  <Form.Group controlId="formGridAddress1">
    <Form.Label>Address</Form.Label>
    <Form.Control className="is-valid" placeholder="1234 Main St" aria-describedby="addressFeedback" />
    <Form.Control.Feedback id="addressFeedback" type="valid">
      This is a nice place!
    </Form.Control.Feedback>
  </Form.Group>

  <Form.Row>
    <Form.Group as={Col} controlId="formGridCity">
      <Form.Label>City</Form.Label>
      <Form.Control />
    </Form.Group>

    <Form.Group as={Col} controlId="formGridState">
      <Form.Label>State</Form.Label>
      <Form.Control as="select" defaultValue="Choose...">
        <option>Choose...</option>
        <option>...</option>
      </Form.Control>
    </Form.Group>

    <Form.Group as={Col} controlId="formGridZip">
      <Form.Label>Zip</Form.Label>
      <Form.Control />
    </Form.Group>
  </Form.Row>

  <Form.Group controlId="formSwitch">
    <Form.Check
      type="switch"
      id="custom-switch"
      label="Check this switch"
    />
    <Form.Check
      disabled
      type="switch"
      label="disabled switch"
      id="disabled-custom-switch"
    />
  </Form.Group>

  <Form.Group id="formGridCheckbox">
    <Form.Check type="checkbox" id="formGridCheckboxCheck" label="Check me out" />
  </Form.Group>

  <Button variant="primary" type="submit">
    Submit
  </Button>
</Form>
```
